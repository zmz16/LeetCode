## 单调栈

```cpp
class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        stack<int> st;
        heights.insert(heights.begin(), 0);   // 数组头部插入0
        heights.push_back(0);   // 数组尾部插入0
        int result = 0;
        st.push(0);
        for (int i = 1; i < heights.size(); i++) {
            while (heights[i] < heights[st.top()]) {
                int mid = st.top();
                st.pop();
                int h = heights[mid];
                int w = i - st.top() - 1;
                result = max(result, h * w);
            }
            st.push(i);
        }
        return result;
    }
};
```
