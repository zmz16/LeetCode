## C++

```cpp
class Solution {
public:
    vector<int> nextGreaterElement(vector<int>& nums1, vector<int>& nums2) {
        stack<int> st;   // 栈，求下一个更大的元素
        vector<int> result(nums1.size(), -1);   // 初始化为-1

        unordered_map<int, int> umap;   // key:下标元素，value：下标
        for (int i = 0; i < nums1.size(); i++) {
            umap[nums1[i]] = i;
        }

        st.push(0);   // 先入栈nums2数组的第一个下标
        for (int i = 1; i < nums2.size(); i++) {
            while (!st.empty() && nums2[i] > nums2[st.top()]) {  // 栈不为空且下一个元素大于栈顶元素
                if (umap.count(nums2[st.top()])) {   // nums1中有栈顶元素
                    result[umap[nums2[st.top()]]] = nums2[i];
                }
                st.pop();
            }
            st.push(i);
        }
        return result;
    }
};
```
