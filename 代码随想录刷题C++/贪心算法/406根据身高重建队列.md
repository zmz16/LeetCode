# 题目链接

https://leetcode.cn/problems/queue-reconstruction-by-height/description/

## 使用数组实现

**有两个维度一定不要同时去考虑**，分开去考虑。

先按照身高排序，然后按照比他高的个数进行插入操作，比如有0个比他高的，插入下标为0的位置，这个操作不会影响后面的k值，因为能插入到前面的说明都比原来的小，不会影响。

```cpp
class Solution {
public:
    static bool cmp(const vector<int>& a, const vector<int>& b) {   // 自定义按照身高排序的函数
        if (a[0] == b[0]) return a[1] < b[1];
        return a[0] > b[0];
    }

    vector<vector<int>> reconstructQueue(vector<vector<int>>& people) {
        sort(people.begin(), people.end(), cmp);   // 先按照身高进行排序
        vector<vector<int>> queue;   // 定义输出结果数组
        for (int i = 0; i < people.size(); i++) {   // 遍历
            int position = people[i][1];   // 记录位置
            queue.insert(queue.begin() + position, people[i]);   // 插入操作
        }
        return queue;
    }
};
```

## 底层使用链表实现

```cpp
class Solution {
public:
    static bool cmp(const vector<int>& a, const vector<int>& b) {   // 自定义按照身高排序的函数
        if (a[0] == b[0]) return a[1] < b[1];
        return a[0] > b[0];
    }

    vector<vector<int>> reconstructQueue(vector<vector<int>>& people) {
        sort(people.begin(), people.end(), cmp);   // 先按照身高进行排序
        list<vector<int>> queue;   // list底层是链表实现，插入效率比vector高的多
        for (int i = 0; i < people.size(); i++) {   // 遍历
            int position = people[i][1];   // 记录位置
            list<vector<int>>::iterator it = queue.begin();
            while (position--) {
                it++;
            }
            queue.insert(it, people[i]);   // 插入操作
        }
        return vector<vector<int>>(queue.begin(), queue.end());
    }
};
```
