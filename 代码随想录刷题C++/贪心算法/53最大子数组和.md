## C++

```cpp
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int ans = INT_MIN;
        int sum = 0;
        for (int i = 0; i < nums.size(); i++) {
            sum += nums[i];
            if (sum > ans) {   // 取区间累计的最大值（相当于不断确定最大子序终止位置）
                ans = sum;
            }
            if (sum <= 0) {   // 相当于重置最大子序起始位置，因为遇到负数一定是拉低总和
                sum = 0;
            }
        }
        return ans;
    }
};
```
