https://leetcode.cn/problems/partition-labels/description/

## C++

```cpp
class Solution {
public:
    vector<int> partitionLabels(string s) {
        int hash[26] = {0};
        for (int i = 0; i < s.size(); i++) {
            hash[s[i] - 'a'] = i;   // 统计某字符最后一位的下标
        }

        vector<int> result;
        int left = 0, right = 0;
        for (int i = 0; i < s.size(); i++) {
            right = max(right, hash[s[i] - 'a']);
            if (i == right) {   // 遍历到达右边界
                result.push_back(right - left + 1);
                left = i + 1;
            }
        }
        return result;
    }
};
```
