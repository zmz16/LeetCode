# 多重背包

https://kamacoder.com/problempage.php?pid=1066

## C++

将每件物品展开就转变为01背包问题，具体是**在遍历背包的for循环里面再加一个遍历物品数量的for循环**。

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    int c = 0, n = 0;
    cin >> c >> n;   // 种类，空间

    vector<int> weight(n);
    vector<int> price(n);
    vector<int> number(n);

    for (int i = 0; i < n; i++) {
        cin >> weight[i];
    }
    for (int i = 0; i < n; i++) {
        cin >> price[i];
    }
    for (int i = 0; i < n; i++) {
        cin >> number[i];
    }
    
    vector<int> dp(c + 1, 0);

    for (int i = 0; i < n; i++) {   // 物品
        for (int j = c; j >= weight[i]; j--) {   // 背包
            for (int k = 1; k <= number[i] && j >= k * weight[i]; k++) {
                dp[j] = max(dp[j], dp[j - k * weight[i]] + k * price[i]);
            }
        }
    }
    cout << dp[c];
    return 0;
}
```
