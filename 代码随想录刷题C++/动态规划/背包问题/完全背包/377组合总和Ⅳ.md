## 一维数组

对于这种排列问题，**先遍历背包，再遍历物品**。

```cpp
class Solution {
public:
    int combinationSum4(vector<int>& nums, int target) {
        vector<unsigned> dp(target + 1, 0);
        dp[0] = 1;
        for (int i = 0; i <= target; i++) {   // target
            for (int j = 0; j < nums.size(); j++) {   // 数组
                 if (i >= nums[j]) {
                    dp[i] += dp[i - nums[j]]; 
                 }
            }
        }
        return dp[target];
    }
};
```
