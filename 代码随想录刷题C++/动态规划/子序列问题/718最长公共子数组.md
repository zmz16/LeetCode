# 题目链接

https://leetcode.cn/problems/maximum-length-of-repeated-subarray/description/

## 自己写的

```cpp
class Solution {
public:
    int findLength(vector<int>& nums1, vector<int>& nums2) {
        vector<vector<int>> dp(nums1.size(), vector<int>(nums1.size(), 0));
        int ans = 0;
        for (int i = 0; i < nums1.size(); i++) {
            for (int j = 0; j < nums2.size(); j++) {
                if (nums1[i] == nums2[j]) {
                    dp[i][j] = 1;
                    ans = 1;   // 打补丁，避免下面的循环漏掉第0行第0列
                }
            }
        }
        for (int i = 1; i < nums1.size(); i++) {
            for (int j = 1; j < nums2.size(); j++) {
                if (dp[i - 1][j - 1] != 0 && dp[i][j] != 0) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                }
                ans = max(ans, dp[i][j]);
            }
        }
        return ans;
    }
};
```

## 别人优化版

```cpp
class Solution {
public:
    int findLength(vector<int>& nums1, vector<int>& nums2) {
        vector<vector<int>> dp(nums1.size() + 1, vector<int>(nums1.size() + 1, 0));
        int ans = 0;
        for (int i = 1; i < nums1.size() + 1; i++) {   // 遍历的是dp数组，而不是传入的参数数组
            for (int j = 1; j < nums2.size() + 1; j++) {
                if (nums1[i - 1] == nums2[j - 1]) {
                    dp[i][j] = dp[i - 1][j - 1] + 1;
                }
                ans = max(ans, dp[i][j]);
            }
        }
        return ans;
    }
};
```
