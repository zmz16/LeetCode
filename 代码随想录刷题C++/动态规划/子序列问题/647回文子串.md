https://leetcode.cn/problems/palindromic-substrings/description/

## 自己写的

```cpp
class Solution {
    bool palidString (string s) {
        for (int i = 0, j = s.size() - 1; i < j; i++, j--) {
            if (s[i] != s[j]) {
                return false;
            }
        }
        return true;
    }
public:
    int countSubstrings(string s) {
        int ans = 0;
        for (int i = 1; i <= s.size(); i++) {   // 必须是i <= s.size()，因为这里的i不是下标，否则遍历不到最后一个字符
            for (int j = 0; j < i; j++) {
                string str = s.substr(j, i - j);
                if (palidString(str)) {
                    ans++;
                }
            }
        }
        return ans;
    }
};
```

## 动态规划

布尔类型的dp[i][j]：表示区间范围[i,j] （注意是左闭右闭）的子串是否是回文子串，如果是dp[i][j]为true，否则为false。

```cpp
class Solution {
public:
    int countSubstrings(string s) {
        vector<vector<bool>> dp(s.size(), vector<bool>(s.size(), false));   // 先初始化为false
        int ans = 0;
        for (int i = s.size() - 1; i >= 0; i--) {
            for (int j = i; j < s.size(); j++) {
                if (s[i] == s[j]) {
                    if (j - i <= 1) {
                        dp[i][j] = true;
                        ans++;
                    }
                    else if (dp[i + 1][j - 1]) {
                        dp[i][j] = true;
                        ans++;
                    }
                }
            }
        }
        return ans;
    }
};
```
