## 二维数组

```cpp
class Solution {
public:
    int change(int amount, vector<int>& coins) {
        // 定义dp数组
        vector<vector<unsigned>> dp(coins.size(), vector<unsigned>(amount + 1, 0));
        // 初始化
        for (int j = 0; j <= amount; j++) {
            if (j % coins[0] == 0) {   // 对于coins[0]，只要背包容量是其整数倍，就有一种方案
                dp[0][j] = 1;
            }
        }
        for (int i = 0; i < coins.size(); i++) {
            dp[i][0] = 1;   // 装满容量为0的背包也有一种方法
        }

        // 遍历
        for (int i = 1; i < coins.size(); i++) {
            for (int j = 1; j <= amount; j++) {
                if (j < coins[i]) {
                    dp[i][j] = dp[i - 1][j];
                }
                else {
                    dp[i][j] = dp[i][j - coins[i]] + dp[i - 1][j];
                }
            }
        }
        return dp[coins.size() - 1][amount];
    }
};
```

## 滚动数组

```cpp
class Solution {
public:
    int change(int amount, vector<int>& coins) {
        // 定义dp数组
        vector<unsigned> dp(amount + 1, 0);
        // 初始化
        dp[0] = 1;

        // 遍历
        for (int i = 0; i < coins.size(); i++) {
            for (int j = coins[i]; j <= amount; j++) {
                dp[j] += dp[j - coins[i]];
            }
        }
        return dp[amount];
    }
};
```
