# 题目链接

https://leetcode.cn/problems/coin-change/description/

## C++

```cpp
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        vector<int> dp(amount + 1, INT_MAX);   // 后面要比较最小值，所以初始化为int类型的最大值
        dp[0] = 0;   // 凑成总金额0需要最少硬币个数是0
        for (int i = 0; i < coins.size(); i++) {   // 硬币
            for (int j = coins[i]; j <= amount; j++) {   // 总金额
                if (dp[j - coins[i]] != INT_MAX) {   // 等于初始值就跳过，说明无法用硬币组成金额 j - coins[i]，例如示例2
                    dp[j] = min(dp[j], dp[j - coins[i]] + 1);
                }
            }
        }
        if (dp[amount] == INT_MAX) {
            return -1;
        }
        return dp[amount];
    }
};
```
