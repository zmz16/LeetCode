## C++

```cpp
class Solution {
public:
    bool canPartition(vector<int>& nums) {
        // 找一个和为总和一半的子集，转换为01背包问题
        int sum = 0;
        for (int i = 0; i < nums.size(); i++) {
            sum += nums[i];
        }
        if (sum % 2 != 0) {   // 奇数
            return false;
        }
        int bagsize = sum / 2;   // 背包大小
        vector<int> dp(bagsize + 1, 0); 
        int j;
        for (int i = 0; i < nums.size(); i++) {
            for (int j = bagsize; j >= nums[i]; j--) {   // j表示背包容量，这里的循环控制条件意思时，只有该数字能放进背包才进入循环，否则没必要
                dp[j] = max(dp[j], dp[j - nums[i]] + nums[i]);
                if (dp[j] == bagsize) {
                    return true;
                }
            }
        }  
        return false;
    }
};
```
