# 题目链接

https://leetcode.cn/problems/implement-queue-using-stacks/description/

## 思路

用两个栈来模拟队列

## C++

```C++
class MyQueue {
public:
    stack<int> stIn;
    stack<int> stOut;

    MyQueue() {
        
    }
    
    void push(int x) {
        // 直接入栈
        stIn.push(x);
    }
    
    int pop() {
        // 当stOut为空的时候，将stIn的所有元素导入stOut
        if (stOut.empty()) {
            while (!stIn.empty()) {
                stOut.push(stIn.top());
                stIn.pop();
            }
        }
        int result = stOut.top();
        stOut.pop();
        return result;
    }
    
    int peek() {
        int result = this->pop();
        stOut.push(result);
        return result;
    }
    
    bool empty() {
        return stIn.empty() && stOut.empty();
    }
};
```
