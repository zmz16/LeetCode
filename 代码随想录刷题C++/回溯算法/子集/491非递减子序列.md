# 题目链接

https://leetcode.cn/problems/non-decreasing-subsequences/description/

与子集问题不同的是，子集收集所有节点，本题收集`path.size() >= 2`的节点。

## 使用set去重

```cpp
class Solution {
private:
    vector<int> path;
    vector<vector<int>> res;

    void backtracking(vector<int>& nums, int startIndex) {
        if (path.size() >= 2) {
            res.push_back(path);
        }

        unordered_set<int> uset;   // 每次递归新建一个集合

        for (int i = startIndex; i < nums.size(); i++) {
            if (uset.find(nums[i]) != uset.end() || (!path.empty() && nums[i] < path.back())) {   // 当前层已使用过该数字，或者当前的nums[i]小于path的最后一个元素
                continue;   // 跳过重复元素
            }
            uset.insert(nums[i]);   // 记录已使用
            path.push_back(nums[i]);
            backtracking(nums, i + 1);
            path.pop_back();
        }
    }
public:
    vector<vector<int>> findSubsequences(vector<int>& nums) {
        path.clear();
        res.clear();
        backtracking(nums, 0);
        return res;
    }
};
```

## 使用数组去重

题目说明`-100 <= nums[i] <= 100`，所以使用数组比使用set更高效。

```cpp
class Solution {
private:
    vector<int> path;
    vector<vector<int>> res;

    void backtracking(vector<int>& nums, int startIndex) {
        if (path.size() >= 2) {
            res.push_back(path);
        }

        int used[201] = {0};   // 每次递归新建一个数组

        for (int i = startIndex; i < nums.size(); i++) {
            if (used[nums[i] + 100] == 1 || (!path.empty() && nums[i] < path.back())) {   // 当前层已使用过该数字，或者当前的nums[i]小于path的最后一个元素
                continue;   // 跳过重复元素
            }
            used[nums[i] + 100] = 1;   // 记录已使用
            path.push_back(nums[i]);
            backtracking(nums, i + 1);
            path.pop_back();
        }
    }
public:
    vector<vector<int>> findSubsequences(vector<int>& nums) {
        path.clear();
        res.clear();
        backtracking(nums, 0);
        return res;
    }
};
```
