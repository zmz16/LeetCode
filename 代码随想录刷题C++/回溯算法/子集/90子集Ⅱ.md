## 自己写的（一次性通过，记录一下）

```cpp
class Solution {
private:
    vector<int> path;
    vector<vector<int>> res;

    void backtracking(vector<int>& nums, int startIndex) {
        res.push_back(path);

        if (startIndex >= nums.size()) {
            return;
        }

        for (int i = startIndex; i < nums.size(); i++) {
            if (i > startIndex && nums[i] == nums[i - 1]) {
                continue;
            }
            path.push_back(nums[i]);
            backtracking(nums, i + 1);
            path.pop_back();
        }
    }
public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        path.clear();
        res.clear();
        sort(nums.begin(), nums.end());
        backtracking(nums, 0);
        return res;
    }
};
```

## 使用set去重

无需依赖排序顺序

```cpp
class Solution {
private:
    vector<int> path;
    vector<vector<int>> res;

    void backtracking(vector<int>& nums, int startIndex) {
        res.push_back(path);

        unordered_set<int> uset;   // 每次递归新建一个集合

        for (int i = startIndex; i < nums.size(); i++) {
            if (uset.find(nums[i]) != uset.end()) {   // 当前层已使用过该数字
                continue;   // 跳过重复元素
            }
            uset.insert(nums[i]);   // 记录已使用
            path.push_back(nums[i]);
            backtracking(nums, i + 1);
            path.pop_back();
        }
    }
public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        path.clear();
        res.clear();
        backtracking(nums, 0);
        return res;
    }
};
```
