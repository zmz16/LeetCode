# 题目链接

## C++

```cpp
class Solution {
public:
    void backtracking(int k, int n, int startIndex, vector<int> & path, vector<vector<int>> & res) {
        /*
        k：子集个数
        n：总和减去处理过的元素，当其为0时满足题意
        startIndex：每次开启新循环的下标
        */
        if (path.size() == k) {   // 先判断长度
            if (n == 0) {   // 如果和满足条件，加入结果集
                res.push_back(path);
            }
            return;   // 只要长度符合条件，都返回
        }
        if (n < 0) {   // 和已经超过目标
            return;
        }

        for (int i = startIndex; i <= 9; i++) {
            path.push_back(i);
            backtracking(k, n - i, i + 1, path, res);
            path.pop_back();
        }
    }

    vector<vector<int>> combinationSum3(int k, int n) {
        int startIndex = 1;
        vector<int> path;
        vector<vector<int>> res;
        backtracking(k, n, startIndex, path, res);
        return res;
    }
};
```

## 剪枝优化

在for循环的控制条件中加一个`sum + candidates[i] <= target`，此时就不需要迭代的终止条件判断`sum > 0`，但该方法实现的前提是要**对数组进行排序**。

```cpp
class Solution {
public:
    void backtracking(vector<int> candidates, int target, int sum, int startIndex, vector<int> & path, vector<vector<int>> & res) {
        if (sum == target) {
            res.push_back(path);
            return;
        }

        // 单层递归的逻辑
        for (int i = startIndex; i < candidates.size() && sum + candidates[i] <= target; i++) {
            path.push_back(candidates[i]);
            sum += candidates[i];
            backtracking(candidates, target, sum, i, path, res);
            sum -= candidates[i];
            path.pop_back();
        }
    }
    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        int sum = 0;
        int startIndex = 0;
        vector<int> path;
        vector<vector<int>> result;
        sort(candidates.begin(), candidates.end());   // 需要排序
        backtracking(candidates, target, sum, startIndex, path, result);
        return result;
    }
};
```
