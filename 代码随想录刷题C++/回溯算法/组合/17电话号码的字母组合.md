## C++

```cpp
class Solution {
private:
    const string letterMap[10] = {
        "",     // 0
        "",     // 1
        "abc",  // 2
        "def",  // 3
        "ghi",  // 4
        "jkl",  // 5
        "mno",  // 6
        "pqrs", // 7
        "tuv",  // 8
        "wxyz", // 9
    };

public:
    void backtracking(string digits, int index, string& s,
                      vector<string>& result) {
        if (digits.size() == index) {
            result.push_back(s);
            return;
        }

        int digit = digits[index] - '0'; // 取输入中的一个字符并将其转换为int类型
        string letter = letterMap[digit]; // 得到数字的字母映射
        for (int i = 0; i < letter.size(); i++) {
            s.push_back(letter[i]);
            backtracking(digits, index + 1, s, result);
            s.pop_back();
        }
    }

    vector<string> letterCombinations(string digits) {
        string s;
        vector<string> result;
        if (digits.size() == 0) {
            return result;
        }
        backtracking(digits, 0, s, result);
        return result;
    }
};
```
