## 暴力回溯

```cpp
class Solution {
public:
    vector<vector<int>> result;
    vector<int> path;

    // 回溯函数，startIndex记录下一层递归的起始位置
    void backtracking(int n, int k, int startIndex) {   // 
        // 终止条件，找到一个大小为k的组合了，返回上一层
        if (path.size() == k) {
            result.push_back(path);  // 增加到结果数组中
            return;
        }

        // 单层搜索的过程
        for (int i = startIndex; i <= n; i++) {
            path.push_back(i);   // 处理节点
            backtracking(n, k, i + 1);   // 递归
            path.pop_back();   // 回溯，撤销处理的节点
        }
    }

    vector<vector<int>> combine(int n, int k) {
        result.clear();   // 可以不写
        path.clear();   // 可以不写
        backtracking(n, k, 1);
        return result;
    }
};
```

## 剪枝优化

在单层搜索过程中，for循环的变化范围`i <= n - (k - path.size()) + 1`。

```cpp
class Solution {
public:
    vector<vector<int>> result;
    vector<int> path;

    // 回溯函数，startIndex记录下一层递归的起始位置
    void backtracking(int n, int k, int startIndex) {   // 
        // 终止条件，找到一个大小为k的组合了，返回上一层
        if (path.size() == k) {
            result.push_back(path);  // 增加到结果数组中
            return;
        }

        // 单层搜索的过程
        for (int i = startIndex; i <= n - (k - path.size()) + 1; i++) {
            path.push_back(i);   // 处理节点
            backtracking(n, k, i + 1);   // 递归
            path.pop_back();   // 回溯，撤销处理的节点
        }
    }

    vector<vector<int>> combine(int n, int k) {
        result.clear();   // 可以不写
        path.clear();   // 可以不写
        backtracking(n, k, 1);
        return result;
    }
};
```

## 复杂度分析

- 时间复杂度: $O(n * 2^n)$
- 空间复杂度: $O(n)$
