## C++

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <unordered_map>
#include <queue>

using namespace std;

int main() {
	int n, m, s, t;   // 文件和依赖关系
	cin >> n >> m;

	vector<int> inDegree(n, 0);   // 记录入度
	unordered_map<int, vector<int>> umap;   // 记录依赖关系
	vector<int> result;   // 记录结果

	while (m--) {
		cin >> s >> t;
		inDegree[t]++;   // 统计入度
		umap[s].push_back(t);   // 记录依赖关系
	}

	// 定义一个队列,存储入度为0的节点,依次进行处理
	queue<int> que;
	for (int i = 0; i < n; i++) {
		if (inDegree[i] == 0) {
			que.push(i);
		}
	}

	// 不断遍历队列的元素
	while (!que.empty()) {
		int cur = que.front();
		que.pop();
		result.push_back(cur);
		vector<int> files = umap[cur];   // //获取该文件指向的文件
		if (files.size()) {    // cur有后续文件
			for (int j = 0; j < umap[cur].size(); j++) {
				inDegree[files[j]]--;
				if (inDegree[files[j]] == 0) {
					que.push(files[j]);
				}
			}
		}
	}

	if (result.size() == n) {
		for (int i = 0; i < n - 1; i++) {
			cout << result[i] << " ";
		}
		cout << result[result.size() - 1] << endl;
	}
	else {
		cout << -1 << endl;
	}
	
	return 0;
}
```
