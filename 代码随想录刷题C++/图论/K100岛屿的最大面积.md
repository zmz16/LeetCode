## 深搜

```cpp
#include <iostream>
#include <vector>

using namespace std;

int dir[4][2] = {0, 1, 1, 0, -1, 0, 0, -1};   // 搜索方向
 
// 深搜
void dfs(vector<vector<int>>& grid, vector<vector<int>>& grid, int x, int y) {
    for (int i = 0; i < 4; i++) {
        int next_x = x + dir[i][0];
        int next_y = y + dir[i][1];
        if (next_x < 0 || next_x >= grid.size() || next_y < 0 || next_y >= grid[0].size()) {   // 访问越界
            continue;
        }
        if (!visited[next_x][next_y] && grid[next_x][next_y] == 1) {
            visited[next_x][next_y] = true;
            dfs(grid, visited, next_x, next_y);
            count++;
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;

    vector<vector<int>> grid(n, vector<int>(m));
    // 读取矩阵
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> grid[i][j];
        }
    }

    vector<vector<int>> visited(n, vector<int>(m));   // 记录访问过的点
    int ans = 1;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (!visited[i][j] && grid[i][j] == 1) {
                int count == 1;
                visited[i][j] = true;
                dfs(grid, visited, i, j);
                ans = max(count, ans);
            }
        }
    }
    cout << ans;
    return 0;
}
```

## 广搜

```cpp
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int dir[4][2] = {0, 1, 1, 0, -1, 0, 0, -1};   // 搜索方向
int count;
 
// 深搜
void bfs(vector<vector<int>>& grid, vector<vector<int>>& visited, int x, int y) {
    queue<pair<int, int>> que;   // 队列存储搜索顺序
    que.push({x, y});
    visited[x][y] = true;

    while (!que.empty()) {
        pair<int, int> cur = que.front();
        que.pop();

        for (int i = 0; i < 4; i++) {
            int next_x = cur.first + dir[i][0];
            int next_y = cur.second + dir[i][1];
            if (next_x < 0 || next_x >= grid.size() || next_y < 0 || next_y >= grid[0].size()) {
                continue;
            }
            if (!visited[next_x][next_y] && grid[next_x][next_y] == 1) {
                visited[next_x][next_y] = true;
                bfs(grid, visited, next_x, next_y);
                count++;
            }
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;

    vector<vector<int>> grid(n, vector<int>(m));
    // 读取矩阵
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> grid[i][j];
        }
    }

    vector<vector<int>> visited(n, vector<int>(m));   // 记录访问过的点
    int ans = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (!visited[i][j] && grid[i][j] == 1) {
                count = 1;   // 遇到陆地时将置为1，剩下的在深搜里进行统计
                visited[i][j] = true;
                bfs(grid, visited, i, j);
                ans = max(count, ans);
            }
        }
    }
    cout << ans;
    return 0;
}
```
