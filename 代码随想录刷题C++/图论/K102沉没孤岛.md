#include <iostream>
#include <vector>

using namespace std;

int dir[4][2] = { 0, 1, 1, 0, -1, 0, 0, -1 };   // 搜索方向

// 深搜，将不是孤岛的陆地置为2
void dfs(vector<vector<int>>& grid, vector<vector<bool>>& visited, int x, int y) {
    grid[x][y] = 2;   // 满足条件了先置为0

    for (int i = 0; i < 4; i++) {
        int next_x = x + dir[i][0];
        int next_y = y + dir[i][1];
        if (next_x < 0 || next_x >= grid.size() || next_y < 0 || next_y >= grid[0].size()) {
            continue;
        }
        if (!visited[next_x][next_y] && grid[next_x][next_y] == 1) {
            visited[next_x][next_y] = true;
            dfs(grid, visited, next_x, next_y);
        }
    }

}

int main() {
    int n, m;
    cin >> n >> m;

    vector<vector<int>> grid(n, vector<int>(m));
    // 读取矩阵
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> grid[i][j];
        }
    }

    vector<vector<bool>> visited(n, vector<bool>(m, false));   // 记录访问过的点

    // 第一列和最后一列
    for (int i = 0; i < n; i++) {
        if (grid[i][0] == 1) {
            visited[i][0] = true;
            dfs(grid, visited, i, 0);
        }
        if (grid[i][m - 1] == 1) {
            visited[i][m - 1] = true;
            dfs(grid, visited, i, m - 1);
        }
    }

    // 第一行和最后一行
    for (int j = 0; j < m; j++) {
        if (grid[0][j] == 1) {
            visited[0][j] = true;
            dfs(grid, visited, 0, j);
        }
        if (grid[n - 1][j] == 1) {
            visited[n - 1][j] = true;
            dfs(grid, visited, n - 1, j);
        }
    }

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (grid[i][j] == 1) {
                grid[i][j] = 0;
            }
            if (grid[i][j] == 2) {
                grid[i][j] = 1;
            }
        }
    }

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cout << grid[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}
