# 题目链接

https://leetcode.cn/problems/all-paths-from-source-to-target/description/

## C++

```cpp
class Solution {
    vector<vector<int>> result;
    vector<int> path;

    void dfs(const vector<vector<int>>& graph, int x, int n) {   // 深度优先搜索
        if (x == n) {
            result.push_back(path);
            return;
        }

        for (int i : graph[x]) {   // 当前搜索节点指向的节点
            path.push_back(i);
            dfs(graph, i, n);
            path.pop_back();
        }
    }
public:
    vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
        result.clear();
        path.clear();

        path.push_back(0);
        dfs(graph, 0, graph.size() - 1);
        return result;
    }
};
```
