## 解法一：直接看陆地周围的情况

```cpp
#include <iostream>
#include <vector>
#include <unordered_map>
#include <unordered_set>

using namespace std;

int main() {
    int n, m;
    cin >> n >> m;

    vector<vector<int>> grid(n, vector<int>(m));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> grid[i][j];
        }
    }

    int dir[4][2] = { 0, 1, 1, 0, -1, 0, 0, -1 };   // 搜索方向
    int result = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (grid[i][j] == 1) {
                for (int k = 0; k < 4; k++) {
                    int x = i + dir[k][0];
                    int y = j + dir[k][1];
                    if (x < 0 || x >= n || y < 0 || y >= m || grid[x][y] == 0) {
                        result++;
                    }
                }
            }
        }
    }
    cout << result << endl;
}
```

## 解法二：result = 岛屿数量 * 4 - cover * 2

cover 是相邻的岛屿数量

```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    int n, m;
    cin >> n >> m;

    vector<vector<int>> grid(n, vector<int>(m, 0));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> grid[i][j];
        }
    }

    int sum = 0;   // 陆地数量
    int cover = 0;   // 相邻数量
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (grid[i][j] == 1) {
                sum++;
                // 统计上面的相邻陆地
                if (i - 1 >= 0 && grid[i - 1][j] == 1) {
                    cover++;
                }
                // 统计左面的相邻陆地
                if (j - 1 >= 0 && grid[i][j - 1] == 1) {
                    cover++;
                }
            }
        }
    }

    cout << sum * 4 - cover * 2 << endl;
}
```
