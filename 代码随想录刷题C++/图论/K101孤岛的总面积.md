# 题目链接

https://kamacoder.com/problempage.php?pid=1173

**深搜或广搜将不符合条件的（与边缘有接触的陆地块）置为0，然后只需要遍历矩阵，统计值为1的元素个数。**

## 深搜

```cpp
#include <iostream>
#include <vector>

using namespace std;

int dir[4][2] = { 0, 1, 1, 0, -1, 0, 0, -1 };   // 搜索方向

// 深搜
void dfs(vector<vector<int>>& grid, vector<vector<bool>>& visited, int x, int y) {
    grid[x][y] = 0;   // 满足条件了先置为0

    for (int i = 0; i < 4; i++) {
        int next_x = x + dir[i][0];
        int next_y = y + dir[i][1];
        if (next_x < 0 || next_x >= grid.size() || next_y < 0 || next_y >= grid[0].size()) {
            continue;
        }
        if (!visited[next_x][next_y] && grid[next_x][next_y] == 1) {
            visited[next_x][next_y] = true;
            dfs(grid, visited, next_x, next_y);
        }
    }
    
}

int main() {
    int n, m;
    cin >> n >> m;

    vector<vector<int>> grid(n, vector<int>(m));
    // 读取矩阵
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> grid[i][j];
        }
    }

    vector<vector<bool>> visited(n, vector<bool>(m, false));   // 记录访问过的点

    // 第一列和最后一列
    for (int i = 0; i < n; i++) {
        if (grid[i][0] == 1) {
            visited[i][0] = true;
            dfs(grid, visited, i, 0);
        }
        if (grid[i][m - 1] == 1) {
            visited[i][m - 1] = true;
            dfs(grid, visited, i, m - 1);
        }
    }

    // 第一行和最后一行
    for (int j = 0; j < m; j++) {
        if (grid[0][j] == 1) {
            visited[0][j] = true;
            dfs(grid, visited, 0, j);
        }
        if (grid[n - 1][j] == 1) {
            visited[n - 1][j] = true;
            dfs(grid, visited, n - 1, j);
        }
    }

    int ans = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (grid[i][j] == 1) {
                ans++;
            }
        }
    }
    cout << ans;
    return 0;
}
```

## 广搜

```cpp
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int dir[4][2] = { 0, 1, 1, 0, -1, 0, 0, -1 };   // 搜索方向

void bfs(vector<vector<int>>& grid, vector<vector<bool>>& visited, int x, int y) {
    grid[x][y] = 0;   // 满足条件了先置为0

    queue<pair<int, int>> que;
    que.push({ x, y });
    visited[x][y] = true;

    while (!que.empty()) {
        pair<int, int> cur = que.front();
        que.pop();
        for (int i = 0; i < 4; i++) {
            int next_x = cur.first + dir[i][0];
            int next_y = cur.second + dir[i][1];
            if (next_x < 0 || next_x >= grid.size() || next_y < 0 || next_y >= grid[0].size()) {
                continue;
            }
            if (!visited[next_x][next_y] && grid[next_x][next_y] == 1) {
                visited[next_x][next_y] = true;
                grid[next_x][next_y] = 0;   // 将访问到的点设为0
                que.push({next_x, next_y});
            }
        }
    }
}

int main() {
    int n, m;
    cin >> n >> m;

    vector<vector<int>> grid(n, vector<int>(m));
    // 读取矩阵
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> grid[i][j];
        }
    }

    vector<vector<bool>> visited(n, vector<bool>(m, false));   // 记录访问过的点

    // 第一列和最后一列
    for (int i = 0; i < n; i++) {
        if (grid[i][0] == 1) {
            visited[i][0] = true;
            bfs(grid, visited, i, 0);
        }
        if (grid[i][m - 1] == 1) {
            visited[i][m - 1] = true;
            bfs(grid, visited, i, m - 1);
        }
    }

    // 第一行和最后一行
    for (int j = 0; j < m; j++) {
        if (grid[0][j] == 1) {
            visited[0][j] = true;
            bfs(grid, visited, 0, j);
        }
        if (grid[n - 1][j] == 1) {
            visited[n - 1][j] = true;
            bfs(grid, visited, n - 1, j);
        }
    }

    int ans = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (grid[i][j] == 1) {
                ans++;
            }
        }
    }
    cout << ans;
    return 0;
}
```
