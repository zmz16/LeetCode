## C++

```cpp
class Solution {
public:
    TreeNode * pre = nullptr;
    int result = INT_MAX;

    void traversal(TreeNode * root) {
        if (root == nullptr) {
            return;
        }

        traversal(root->left);

        if (pre != nullptr) {
            result = min(result, root->val - pre->val);
        }
        pre = root;

        traversal(root->right);
    }
    

    int getMinimumDifference(TreeNode* root) {
        traversal(root);
        return result;
    }
};
```
