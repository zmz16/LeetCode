https://leetcode.cn/problems/path-sum/description/

## C++

```cpp
class Solution {
private:
    bool traversal(TreeNode* node, int num) {   // num 表示目标值减去当前节点的值，当遇到叶子节点且num为0时，即找到路径
        if (node->left == nullptr && node->right == nullptr && num == 0) {   // 满足条件时，返回true
            return true;
        }
        if (node->left == nullptr && node->right == nullptr && num != 0) {   // 当前路径不满足条件时，返回false进入上一层
            return false;
        }

        if (node->left) {   // 如果判断node为空，上面的判断会出现操作空指针异常
            // 有一个回溯的过程，这是简化的版本
            if (traversal(node->left, num - node->left->val)) {   // 如果找到路径，直接返回到根节点
                return true;
            }
        }
        if (node->right) {
           if (traversal(node->right, num - node->right->val)) {   // 如果找到路径，直接返回到根节点
                return true;
            }
        }
        return false;
    }
public:
    bool hasPathSum(TreeNode* root, int targetSum) {
        if (root == nullptr) {
            return false;
        }
        return traversal(root, targetSum - root->val);
    }
};
```
