## C++

中序遍历，判断遍历结果是否递增

```cpp
class Solution {
public:
    TreeNode * pre = nullptr;   // 使用成员变量保存前驱节点
    
    bool isValidBST(TreeNode* root) {
        if (root == nullptr) {
          return true;
        }

        bool left = isValidBST(root->left);  // 左

        if (pre != nullptr && pre->val >= root->val) {   // 中
          return false;
        }
        pre = root;

        bool right = isValidBST(root->right);   // 右

        return left && right;
    }
};
```
