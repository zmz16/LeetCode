## C++

class Solution {
public:
    vector<vector<int>> levelOrderBottom(TreeNode* root) {
        vector<vector<int>> result;   // 保存结果
        queue<TreeNode*> que;

        if (root) {
            que.push(root);
        }

        while ( ! que.empty()) {
            int size = que.size();
            vector<int> vec;   // 存储当前层的节点数据
            while (size--) {
                TreeNode * node = que.front();   // 取出队首元素
                que.pop();
                vec.push_back(node->val);

                if (node->left) {
                    que.push(node->left);
                }
                if (node->right) {
                    que.push(node->right);
                }
            }
            result.push_back(vec);
        }
        reverse(result.begin(), result.end());
        return result;
    }
};
