## 递归法

```cpp
class Solution {
public:
    TreeNode* searchBST(TreeNode* root, int val) {
        if (root == nullptr || root->val == val) {
            return root;
        }

        TreeNode * result = nullptr;

        if (val > root->val) {
            result = searchBST(root->right, val);
        }
        else {
            result = searchBST(root->left, val);
        }

        return result;
    }
};
```

## 迭代

```cpp
class Solution {
public:
    TreeNode* searchBST(TreeNode* root, int val) {
        while (root != nullptr) {
            if (val == root->val) {
                return root;
            }
            else if (val > root->val) {
                root = root->right;
            }
            else {
                root = root->left;
            }
        }
        return root;
    }
};
```
