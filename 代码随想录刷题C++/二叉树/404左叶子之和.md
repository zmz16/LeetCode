// 自己写的
class Solution {
public:
    void llSum(TreeNode * curr, int & sum) {
        if (curr == nullptr) {
            return;
        }
        if (curr->left != nullptr && curr->left->left == nullptr && curr->left->right == nullptr) {  // 左叶子节点
            sum += curr->left->val;
        }
        llSum(curr->left, sum);
        llSum(curr->right, sum);
    }
    int sumOfLeftLeaves(TreeNode* root) {
        int sum = 0;
        llSum(root, sum);
        return sum;
    }
};

// 别人简化版本
class Solution {
public:
    int sumOfLeftLeaves(TreeNode* root) {
        int sum = 0;
        if (root == nullptr) {
            return 0;
        }
        if (root->left && !root->left->left && !root->left->right) {
            sum = root->left->val;
        }
        return sum + sumOfLeftLeaves(root->left) + sumOfLeftLeaves(root->right);
    }
};
