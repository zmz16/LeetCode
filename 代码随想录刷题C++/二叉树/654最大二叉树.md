// 自己写的
class Solution {
public:
    TreeNode * traversal(vector<int> nums, int left, int right) {
        if (right - left == 1) {   // 数组只有1个元素，直接构造二叉树
            return new TreeNode(nums[left]);
        }

        int maxIndex = left;
        for (int i = left + 1; i < right; i++) {
            if (nums[i] > nums[maxIndex]) {
                maxIndex = i;
            }
        }

        // 构造根节点
        TreeNode * root = new TreeNode(nums[maxIndex]);

        // 递归
        if (maxIndex > left) {
            root->left = traversal(nums, left, maxIndex);   // 左闭右开
        }
        if (maxIndex < right - 1) {
            root->right = traversal(nums, maxIndex + 1, right);
        }
        
        return root;
    }

    TreeNode* constructMaximumBinaryTree(vector<int>& nums) {
        return traversal(nums, 0, nums.size());
    }
};

// 允许空节点进入递归
class Solution {
public:
    TreeNode * traversal(vector<int> nums, int left, int right) {
        if (left >= right) {   // 空数组，直接返回空节点
            return nullptr;
        }

        // 找最大值下标
        int maxIndex = left;
        for (int i = left + 1; i < right; i++) {
            if (nums[i] > nums[maxIndex]) {
                maxIndex = i;
            }
        }

        // 构造根节点
        TreeNode * root = new TreeNode(nums[maxIndex]);

        // 递归
        root->left = traversal(nums, left, maxIndex);   // 左闭右开
        root->right = traversal(nums, maxIndex + 1, right);
        
        return root;
    }

    TreeNode* constructMaximumBinaryTree(vector<int>& nums) {
        return traversal(nums, 0, nums.size());
    }
};
