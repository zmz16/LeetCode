## C++

```cpp
class Solution {
public:
    // 定义全局变量
    vector<int> vec;
    vector<vector<int>> res;

    // 递归函数不需要返回值，因为要遍历整棵树
    void traversal(TreeNode * node, int sum) {
        if (node->left == nullptr && node->right == nullptr && sum == 0) {
            res.push_back(vec);
        }
        if (node->left) {
            vec.push_back(node->left->val);
            sum -= node->left->val;
            traversal(node->left, sum);
            sum += node->left->val;
            vec.pop_back();
        }
        if (node->right) {
            vec.push_back(node->right->val);
            sum -= node->right->val;
            traversal(node->right, sum);
            sum += node->right->val;
            vec.pop_back();
        }
    }

    vector<vector<int>> pathSum(TreeNode* root, int targetSum) {
        vec.clear();
        res.clear();
        if (root == nullptr) {
            return res;
        }
        vec.push_back(root->val);
        traversal(root, targetSum - root->val);
        vec.pop_back();
        return res;
    }
};
```
