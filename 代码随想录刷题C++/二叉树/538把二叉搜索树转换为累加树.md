## C++

```cpp
class Solution {
public:
    int pre = 0;

    void traveral(TreeNode* node) {
        if (node == nullptr) {
            return;
        }
        traveral(node->right);
        node->val += pre;
        pre = node->val;
        traveral(node->left);
    }

    TreeNode* convertBST(TreeNode* root) {
        traveral(root);
        return root;
    }
};
```
