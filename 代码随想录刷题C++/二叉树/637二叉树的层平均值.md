## C++

```cpp
class Solution {
public:
    vector<double> averageOfLevels(TreeNode* root) {
        vector<double> result;
        queue<TreeNode*> que;

        if (root) {
            que.push(root);
        }

        while ( ! que.empty()) {
            int size = que.size();
            double sum = 0;
            for (int i = 0; i < size; i++) {
                TreeNode * node = que.front();
                que.pop();

                sum += node->val;

                if (node->left) {
                    que.push(node->left);
                }
                if (node->right) {
                    que.push(node->right);
                }
            }
            double avg = sum / size;
            result.push_back(avg);
        }

        return result;
    }
};
```
