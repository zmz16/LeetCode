# 题目链接

https://leetcode.cn/problems/find-bottom-left-tree-value/description/

## 迭代

层序遍历记录每一层第一个节点，直到遍历结束直接返回即为结果。

```cpp
class Solution {
public:
    int findBottomLeftValue(TreeNode* root) {
        queue<TreeNode*> que;
        int result;
        if (root == nullptr) {
            return 0;
        }
        que.push(root);
        while (!que.empty()) {
            int size = que.size();
            for (int i = 0; i < size; i++) {
                TreeNode * node = que.front();
                que.pop();
                if (i == 0) {
                    result = node->val;
                }
                if (node->left) {
                    que.push(node->left);
                }
                if (node->right) {
                    que.push(node->right);
                }
            }
        }
        return result;
    }
};
```

## 迭代

```cpp
class Solution {
public:
    int maxDepth = INT_MIN;
    int result;

    void traversal (TreeNode * curr, int depth) {
        if (curr->left == nullptr && curr->right == nullptr) {
            if (depth > maxDepth) {
                maxDepth = depth;
                result = curr->val;
            }
            return;
        }
        if (curr->left) {
            depth++;   // 深度加一
            traversal(curr->left, depth);
            depth--;   // 回溯，深度减一
        }
        /*
        上面内容与下面逻辑等同，因为不是引用传参，所以递归不改变当前的depth的值
        if (curr->left) {
            traversal(curr->left, depth + 1);
        }
        */
        if (curr->right) {
            depth++;
            traversal(curr->right, depth);
            depth--;
        }
        return;
    }

    int findBottomLeftValue(TreeNode* root) {
        traversal(root, 0);
        return result;
    }
};
```
